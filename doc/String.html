def encode_smart_quotes
  
  # A bit of a hack but this is the best that I can come up with at this time.
  begining_double_smart_quotes_replace = self.gsub(%r([\s]\")/, " &#8220;")
  end_double_smart_quotes_replace = begining_double_smart_quotes_replace.gsub(%r(\"[\s])/, "&#8221; ")
  apostrophe_replace = end_double_smart_quotes_replace.gsub(%r([\S]\'[\S])/) {|chars| chars[0] + '&#8217;' + chars[2]}
  begining_single_smart_quotes_replace = apostrophe_replace.gsub(%r([\s]\')/, " &#8216;")
  end_single_smart_quotes_replace = begining_single_smart_quotes_replace.gsub(%r(\"[\s])/, "&#8217; ")

end

def get_gender
      
  case self.downcase
  when %r^women/
    return "women"
  when %r^men/
    return "men"
  when %r^boy/
    return "boy"
  when %r^girl/
    return "girl"
  when %r^baby/
    return "baby"
  else
    return "unisex"
  end
  
end

def remove_PLM_corruption
  
  # \023 should be an en dash. But it is getting corrupted by the PLM
  clean_string = self.gsub(%r\0023/, "&#8211;")
  
  # \031 should be a ' dash. But it is getting corrupted by the PLM
  clean_string = clean_string.gsub(%r\0031/, "&#8217;")
  
  # \035 should be a double grave mark (or what Edit calls the "inch mark"). But it is getting corrupted by the PLM
  clean_string = clean_string.gsub(%r\0035/, "&#733;")
  
   # \024 Not sure what this should be?? Replacing it with a <space> 
  clean_string = clean_string.gsub(%r\0024/, " ")
  
end

def remove_trade_reg_marks
  
  # Remove Registered Trademarks (R)
  clean_string = self.gsub(%r\&\#174;/, '')
  
  # Remove Trade Marks (TM)
  #clean_string = clean_string.gsub(/\342\204\242/, '')
  clean_string = clean_string.gsub(%r\&\#8482;/, '')
  return clean_string
  
end

def titleCase
   
    words = self.scan(%r[\w']+/)
    newString = String.new
    
    words.each do | word |
      newString += word.capitalize +  " "
    end
    
    newString = newString.strip
    return newString
    
end

def truncate_gender
  m_w = self.gsub(%rWomen's/, "W&#8217;s")
  m_w = m_w.gsub(%rWomen&#8217;s/, "W&#8217;s")
  m_w = m_w.gsub(%rMen&#8217;s/, "M&#8217;s")
  m_w = m_w.gsub(%rMen's/, "M&#8217;s")
  return m_w
end
