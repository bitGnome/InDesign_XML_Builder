def upf(value)
  
  value = test_for_nil(value)
  
  value.downcase!
  case
  when value.match(%r50\+/)
    return @path[:upf_50_plus]
  when value.match(%r50/)
    return @path[:upf_50]
  when value.match(%r40/)
    return @path[:upf_40]
  when value.match(%r35/)
    return @path[:upf_35]
  when value.match(%r30/)
    return @path[:upf_30]
  when value.match(%r25/)
    return @path[:upf_25]
  when value.match(%r20/)
    return @path[:upf_20]
  when value.match(%r15/)
    return@path[:upf_15]
  else
    return nil
  end
  
end
